# Furniture Pipeline — Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Next.js 14 frontend — submit page, job history, and live results page with stage progress, crop gallery, 3D viewer, and render gallery — connecting to the FastAPI backend from Plan 1.

**Architecture:** Next.js 14 App Router SPA. Three pages share a single `api.ts` client. The results page opens an SSE connection to `GET /jobs/{id}/stream` and progressively renders sections as `stage_completed` events arrive. UI components are built with shadcn/ui (Radix + Tailwind). The 3D viewer uses `@google/model-viewer` via a client-side wrapper.

**Tech Stack:** Next.js 14 (App Router), React 18, TypeScript, Tailwind CSS, shadcn/ui, `@google/model-viewer` (web component), native `EventSource` API for SSE.

**Prerequisite:** Plan 1 (backend) must be deployed and reachable at `NEXT_PUBLIC_API_URL` (defaults to `http://localhost:8000`).

## Global Constraints

- TypeScript strict mode everywhere
- Tailwind CSS + shadcn/ui for all styling — no CSS modules, no inline styles (except where `style=` is required by web components)
- `NEXT_PUBLIC_API_URL` environment variable controls the API base URL (set in `.env.local` and in docker-compose)
- `@google/model-viewer` is a web component — import it in a `'use client'` component with `useEffect` to avoid SSR issues
- Next.js App Router — no `pages/` directory
- No server actions, no server components making API calls — all data fetching is client-side (`'use client'`) to keep SSE and polling simple
- Node.js 20 (LTS) in the Docker image
- shadcn/ui component source files live in `src/components/ui/` — never modify them directly; extend via wrapper components in `src/components/`

```ts
// frontend/src/lib/api.ts
import type { JobDetail, JobSummary } from "./types";

const BASE = process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:8000";

async function _json<T>(res: Response): Promise<T> {
  if (!res.ok) {
    const text = await res.text();
    throw new Error(`API ${res.status}: ${text}`);
  }
  return res.json() as Promise<T>;
}

export async function createJobFromUrl(url: string): Promise<{ id: string; status: string }> {
  const form = new FormData();
  form.append("url", url);
  return _json(await fetch(`${BASE}/jobs`, { method: "POST", body: form }));
}

export async function createJobFromImage(file: File): Promise<{ id: string; status: string }> {
  const form = new FormData();
  form.append("image", file);
  return _json(await fetch(`${BASE}/jobs`, { method: "POST", body: form }));
}

export async function listJobs(): Promise<JobSummary[]> {
  return _json(await fetch(`${BASE}/jobs`));
}

export async function getJob(id: string): Promise<JobDetail> {
  return _json(await fetch(`${BASE}/jobs/${id}`));
}

export function openJobStream(id: string): EventSource {
  return new EventSource(`${BASE}/jobs/${id}/stream`);
}
```

```tsx
"use client";
// frontend/src/components/SubmitForm.tsx
import { useRouter } from "next/navigation";
import { useRef, useState } from "react";
import { Alert, AlertDescription } from "@/components/ui/alert";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Input } from "@/components/ui/input";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { createJobFromImage, createJobFromUrl } from "@/lib/api";

export default function SubmitForm() {
  const router = useRouter();
  const [url, setUrl] = useState("");
  const [file, setFile] = useState<File | null>(null);
  const [dragging, setDragging] = useState(false);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const fileRef = useRef<HTMLInputElement>(null);

  async function handleSubmit(tab: "url" | "image") {
    setError(null);
    setLoading(true);
    try {
      let result: { id: string };
      if (tab === "url") {
        if (!url.trim()) throw new Error("Please enter a URL");
        result = await createJobFromUrl(url.trim());
      } else {
        if (!file) throw new Error("Please select an image");
        result = await createJobFromImage(file);
      }
      router.push(`/jobs/${result.id}`);
    } catch (err) {
      setError(err instanceof Error ? err.message : "Submission failed");
      setLoading(false);
    }
  }

  function handleDrop(e: React.DragEvent) {
    e.preventDefault();
    setDragging(false);
    const dropped = e.dataTransfer.files[0];
    if (dropped) setFile(dropped);
  }

  return (
    <Card className="mx-auto w-full max-w-xl">
      <CardHeader>
        <CardTitle>Generate 3D Model</CardTitle>
      </CardHeader>
      <CardContent>
        <Tabs defaultValue="url">
          <TabsList className="mb-6 w-full">
            <TabsTrigger value="url" className="flex-1">Paste URL</TabsTrigger>
            <TabsTrigger value="image" className="flex-1">Upload Image</TabsTrigger>
          </TabsList>

          <TabsContent value="url">
            <div className="space-y-4">
              <Input
                type="url"
                placeholder="https://example.com/furniture-product"
                value={url}
                onChange={(e) => setUrl(e.target.value)}
                onKeyDown={(e) => e.key === "Enter" && handleSubmit("url")}
              />
              {error && (
                <Alert variant="destructive">
                  <AlertDescription>{error}</AlertDescription>
                </Alert>
              )}
              <Button
                className="w-full"
                disabled={loading}
                onClick={() => handleSubmit("url")}
              >
                {loading ? "Submitting…" : "Generate 3D Model"}
              </Button>
            </div>
          </TabsContent>

          <TabsContent value="image">
            <div className="space-y-4">
              <div
                onDragOver={(e) => { e.preventDefault(); setDragging(true); }}
                onDragLeave={() => setDragging(false)}
                onDrop={handleDrop}
                onClick={() => fileRef.current?.click()}
                className={`flex cursor-pointer flex-col items-center justify-center rounded-md border-2 border-dashed py-12 transition-colors ${
                  dragging ? "border-primary bg-accent" : "border-input hover:border-ring"
                }`}
              >
                <input
                  ref={fileRef}
                  type="file"
                  accept="image/png,image/jpeg,image/webp,image/avif"
                  className="hidden"
                  onChange={(e) => setFile(e.target.files?.[0] ?? null)}
                />
                {file ? (
                  <p className="text-sm font-medium">{file.name}</p>
                ) : (
                  <>
                    <p className="text-sm text-muted-foreground">Drop an image here, or click to browse</p>
                    <p className="mt-1 text-xs text-muted-foreground">PNG, JPG, WEBP, AVIF</p>
                  </>
                )}
              </div>
              {error && (
                <Alert variant="destructive">
                  <AlertDescription>{error}</AlertDescription>
                </Alert>
              )}
              <Button
                className="w-full"
                disabled={loading}
                onClick={() => handleSubmit("image")}
              >
                {loading ? "Submitting…" : "Generate 3D Model"}
              </Button>
            </div>
          </TabsContent>
        </Tabs>
      </CardContent>
    </Card>
  );
}
```

```tsx
"use client";
// frontend/src/app/jobs/[id]/page.tsx
import { use, useEffect, useRef, useState } from "react";
import { Alert, AlertDescription } from "@/components/ui/alert";
import { Card, CardContent } from "@/components/ui/card";
import { Separator } from "@/components/ui/separator";
import CropGallery from "@/components/CropGallery";
import DownloadPanel from "@/components/DownloadPanel";
import JobStatusBadge from "@/components/JobStatusBadge";
import PipelineProgress from "@/components/PipelineProgress";
import RenderGallery from "@/components/RenderGallery";
import { getJob, openJobStream } from "@/lib/api";
import type { JobDetail, StageCompletedEvent, StageStartedEvent } from "@/lib/types";

export default function JobPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = use(params);
  const [job, setJob] = useState<JobDetail | null>(null);
  const [completedStages, setCompletedStages] = useState<Set<string>>(new Set());
  const [runningStage, setRunningStage] = useState<string | null>(null);
  const [liveAssets, setLiveAssets] = useState<{
    cropUrls: string[];
    meshTexturedGlbUrl: string | null;
    renders: { front: string | null; side: string | null; top: string | null; angled: string | null };
  }>({ cropUrls: [], meshTexturedGlbUrl: null, renders: { front: null, side: null, top: null, angled: null } });

  // Load initial job state
  useEffect(() => {
    getJob(id).then(setJob);
  }, [id]);

  // SSE subscription
  useEffect(() => {
    if (!job) return;
    if (["completed", "failed", "cancelled"].includes(job.status)) return;

    const es = openJobStream(id);

    es.addEventListener("stage_started", (e: MessageEvent) => {
      const data = JSON.parse(e.data) as StageStartedEvent;
      setRunningStage(data.stage);
    });

    es.addEventListener("stage_completed", (e: MessageEvent) => {
      const data = JSON.parse(e.data) as StageCompletedEvent;
      setCompletedStages((prev) => new Set([...prev, data.stage]));
      setRunningStage(null);

      if (data.stage === "s2_crop" && data.assets.crops) {
        setLiveAssets((prev) => ({ ...prev, cropUrls: data.assets.crops! }));
      }
      if (data.stage === "s5_texture" && data.assets.mesh_textured_glb) {
        setLiveAssets((prev) => ({ ...prev, meshTexturedGlbUrl: data.assets.mesh_textured_glb! }));
      }
      if (data.stage === "s6_render") {
        setLiveAssets((prev) => ({
          ...prev,
          renders: {
            front: data.assets.front ?? null,
            side: data.assets.side ?? null,
            top: data.assets.top ?? null,
            angled: data.assets.angled ?? null,
          },
        }));
      }
    });

    es.addEventListener("job_completed", () => {
      setRunningStage(null);
      es.close();
      getJob(id).then(setJob);
    });

    es.addEventListener("job_failed", (e: MessageEvent) => {
      const data = JSON.parse(e.data);
      setRunningStage(null);
      es.close();
      setJob((prev) => prev ? { ...prev, status: "failed", error_message: data.error } : prev);
    });

    return () => { es.close(); };
  }, [id, job?.status]);

  if (!job) {
    return <p className="text-muted-foreground">Loading…</p>;
  }

  // Merge live SSE assets with final job detail
  const cropUrls = liveAssets.cropUrls.length > 0 ? liveAssets.cropUrls : job.crop_urls;
  const meshGlbUrl = liveAssets.meshTexturedGlbUrl ?? job.mesh_textured_glb_url;
  const renders = {
    front:  liveAssets.renders.front  ?? job.render_front_url,
    side:   liveAssets.renders.side   ?? job.render_side_url,
    top:    liveAssets.renders.top    ?? job.render_top_url,
    angled: liveAssets.renders.angled ?? job.render_angled_url,
  };

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-start justify-between">
        <div>
          <h1 className="text-2xl font-semibold">Job Results</h1>
          <p className="mt-1 font-mono text-xs text-muted-foreground">{id}</p>
          {job.input_url && (
            <p className="mt-1 max-w-md truncate text-sm text-muted-foreground">{job.input_url}</p>
          )}
        </div>
        <JobStatusBadge status={job.status} />
      </div>

      <Separator />

      {/* Metadata */}
      {(job.furniture_category || job.dimensions_mm) && (
        <Card>
          <CardContent className="flex gap-8 pt-4 text-sm">
            {job.furniture_category && (
              <div>
                <p className="text-xs text-muted-foreground">Category</p>
                <p className="font-medium capitalize">{job.furniture_category}</p>
              </div>
            )}
            {job.dimensions_mm && (
              <div>
                <p className="text-xs text-muted-foreground">Dimensions</p>
                <p className="font-medium">
                  {job.dimensions_mm.width_mm} × {job.dimensions_mm.depth_mm} × {job.dimensions_mm.height_mm} mm
                </p>
              </div>
            )}
          </CardContent>
        </Card>
      )}

      {/* Progress */}
      <PipelineProgress completedStages={completedStages} runningStage={runningStage} />

      {/* Error */}
      {job.error_message && (
        <Alert variant="destructive">
          <AlertDescription>{job.error_message}</AlertDescription>
        </Alert>
      )}

      {/* 3D Viewer — placeholder, replaced in Task 6 */}
      {meshGlbUrl && (
        <Card>
          <CardContent className="pt-4">
            <p className="text-sm text-muted-foreground">[3D viewer added in Task 6]</p>
          </CardContent>
        </Card>
      )}

      {/* Crops */}
      <CropGallery cropUrls={cropUrls} />

      {/* Renders */}
      <RenderGallery renders={renders} />

      {/* Downloads */}
      {job.status === "completed" && <DownloadPanel job={job} />}
    </div>
  );
}
```
